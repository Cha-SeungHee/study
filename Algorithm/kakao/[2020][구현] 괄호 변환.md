import java.util.*;

class Solution {
    public String solution(String p) {        
        return convert(p);
    }
    
    private String convert(String str) {
        if (str.length() == 0 || str == null) return "";        
        
        String u = "", v = "", result = "";        
        int balance = 0;
        
        for (int i = 0; i < str.length(); i++) {
            if (i > 0 && balance == 0) {
                u = str.substring(0, i);
                v = str.substring(i);
                
                break;
            }
            
            char ch = str.charAt(i);
            
            if (ch == '(') {
                balance = balance + 1;
            } else if (ch == ')') {
                balance = balance - 1;
            }
        }       
        
        if (balance == 0 && u.length() == 0) u = str.substring(0);
        
        if (isValid(u)) {
            result = u + convert(v);
        } else {
            result = "(";
            result +=  convert(v);
            result += ")";            
            
            String tmp = u.substring(1, u.length() - 1);   
            tmp = swapBracket(tmp);        
            
            result += tmp;            
        }
        
        
        
        return result;
    }
    
    private boolean isValid(String str) {                
        Stack<Character> stack = new Stack<>();
        
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
            
            if (ch == '(') {
                stack.push(ch);
            } else if (ch == ')') {
                if (stack.isEmpty()) return false;
                
                stack.pop();
            }
        }
        
        return true;
    }
    
    private String swapBracket(String str) {
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == '(') {
                sb.append(')');
            } else {
                sb.append('(');
            }     
        }
        
        return sb.toString();
    }
}
