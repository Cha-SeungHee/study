class Solution {
    public String[] solution(int n, int[] arr1, int[] arr2) {
        String[] password = new String[n];
        
        for (int i = 0; i < n; i++) {
            StringBuilder sb = new StringBuilder();
            
            int temp = arr1[i] | arr2[i];
            
            int digit = 0;
            while (digit < n) {
                String letter = ((temp & 1) == 1) ? "#" : " ";
                sb.append(letter);
                
                temp = temp >> 1;
                digit = digit + 1;
            }
            
            password[i] = sb.reverse().toString();
        }        
        
        return password;
    }
}
