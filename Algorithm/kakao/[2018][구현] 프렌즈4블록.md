import java.util.*;

class Solution {
    static int xLen, yLen;
    static char[][] map;
    static boolean[][] popMap;
    static HashSet<Integer> poppedLine;
    static int popCount = 0;
    static boolean popped = true;
    
    static int[] dx = {1, 1, 0}; // right, right-down, down
    static int[] dy = {0, 1, 1};
        
    public int solution(int m, int n, String[] board) {
        yLen = board.length;
        xLen = board[0].length();
        
        map = new char[yLen][xLen];        
        
        for (int j = 0; j < yLen; j++) {
            map[j] = board[j].toCharArray();
        }        
        
        while (popped) {
            checkCondition();
            pop();
            updateMap();
        }
        
        return popCount;
    }
    
    private void checkCondition() {        
        popped = false;
        popMap = new boolean[yLen][xLen];        
        
        for (int j = 0; j < yLen - 1; j++) {
            for (int i = 0; i < xLen - 1; i++) {
                boolean poppable = true;
                char ch = map[j][i];
                
                if (ch == '-') continue;
                
                for (int k = 0; k < 3; k++) {
                    int ny = j + dy[k];
                    int nx = i + dx[k];
                    
                    if (map[ny][nx] != ch) {
                        poppable = false;
                        break;
                    }
                }
                
                if (poppable) {                    
                    popped = true;
                    
                    popMap[j][i] = true;
                    for (int k = 0; k < 3; k++) {
                        int ny = j + dy[k];
                        int nx = i + dx[k];

                        popMap[ny][nx] = true;
                    }
                }
            }
        }        
    }
    
    private void pop() {
        poppedLine = new HashSet<>();
        
        for (int j = 0; j < yLen; j++) {
            for (int i = 0; i < xLen; i++) {
                poppedLine.add(i);
                
                if (popMap[j][i] == true) {
                    map[j][i] = '-';   
                    popCount = popCount + 1;
                }
            }
        }
    }
    
    private void updateMap() {
        for (int line : poppedLine) {
            for (int j = yLen - 1; j >= 1; j--) {
                if (map[j][line] == '-') {
                    int y = j - 1;
                    
                    while (y >= 0) {
                        if (map[y][line] != '-') {
                            map[j][line] = map[y][line];
                            map[y][line] = '-';
                            
                            break;
                        }
                        
                        y = y - 1;
                    }
                }
            }
        }
    }
}
